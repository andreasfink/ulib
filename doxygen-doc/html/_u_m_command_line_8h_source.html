10 
11 UMCommandLine is a helper object for a command line parsing
12 It can be used independently of ulib if needed.
13 
14  Here is an example on how to use it:
15 
16  int main(int argc, const char *argv[])
17  {
18  // we define some general stuff about this app
19  // its used by help and other diagnostic outputs
20  NSDictionary *appDefinition = @ {
21  @"version" : @"1.0",
22  @"executable" : @"myutil",
23  @"run-as" : @(argv[0]),
24  @"copyright" : @"(c) 2018 bla bla",
25  @"param-definition" : @"{input-file}",
26  };
27 
28  // we define a dictionary specifying what option flags and options are allowed
29  // and what parameters they take
30  // the following keys exist
31  // name : mandatory unique internal name of option
32  // short: optional short version -x
33  // long: optional long version --xxxxx
34  // help: mandatory description of the help output
35  // argument: optional naming of the argument. If present, the flag
36  // requires an optional parameter
37  //
38  // options with parameters can be called like this
39  // -x argument
40  // --xlong argument
41  // --xlong=argument
42  // if multipe items:
43  // -x argument1 -x argument2
44  // --xlong argument1 --xlong argument2
45  // --xlong=argument1 --xlong=argument2
46  // --xlong=argument1,argument2
47  //
48  // note: --xlong argument1,argument2 would result in a single
49  // argument with value "argument1,argument2"
50 
51  NSArray *commandLineDefinition = @[
52  @{
53  @"name" : @"version",
54  @"short" : @"-V",
55  @"long" : @"--version",
56  @"help" : @"shows the software version"
57  },
58  @{
59  @"name" : @"verbose",
60  @"short" : @"-v",
61  @"long" : @"--verbose",
62  @"help" : @"enables verbose mode"
63  },
64  @{
65  @"name" : @"help",
66  @"short" : @"-h",
67  @"long" : @"--help",
68  @"help" : @"shows the help screen",
69  }];
70  @{
71  @"name" : @"debug",
72  @"short" : @"-h",
73  @"long" : @"--help",
74  @"argument" : @"debug-option",
75  @"multi" : @(YES),
76  @"help" : @"shows the help screen",
77  }]; // initialize the command line object
78  UMCommandLine *cmd = [[UMCommandLine alloc]initWithCommandLineDefintion:commandLineDefinition
79  appDefinition:appDefinition
80  argc:argc
81  argv:argv];
82 
83  // takes care --help and --version
84  [cmd handleStandardArguments];
85 
86  ...
87  at this point the cmd.params will contain a dictionary of all options
88  (the key is the name from the definiton).
89  for options with arguments the value is an array of the passed arguments.
90  for options without arguments, the value is a NSValue representing the
91  number of times the parameter appeared for example if you pass -vvvv
92  you would have @{ "verbose" : 4 }
93  cmd.mainArguments would contain all parameters you pass without options.
94  if you pass -- then all following strings are considered mainArugments.
95  */
